### We make a plotting function, which can further help with outlier detection

def plot_features(df, features, logx=False, logy=False):
    # initial plt figure settings
    plt.figure(figsize = (9, 6))
    #recasting features for indexing, to support tuple as input
    features = list(features)
    f1 = df[features[0]]
    f2 = df[features[1]]
    if logx:
        plt.xscale("log")
        features[0] = features[0] + " (log10)"
    if logy:
        plt.yscale("log")
        features[1] = features[1] + " (log10)"
    if len(features) == 2:
        plt.scatter(x=f1, y=f2)
    else:
        f3 = df[features[2]]
        for i in range(len(f3)):
            if f3[i] == 0:
                plt.scatter(x = f1[i], y = f2[i], color="b")
            else:
                plt.scatter(x = f1[i], y = f2[i], marker='x')
    
    plt.xlabel(features[0])
    plt.ylabel(features[1])

    plt.show()

#outlier removal with a threshold function
def remove_outliers(df, feature=None, q = 0.99):
    q = df[feature].quantile(q)
    return df[df[feature]< q]

### We create a graphing function that can evalute the effect of k in KBest on basic fit score of classifiers

def graph_k_accuracy(kvalues = 1, clfs = []):
    for clf in clfs:
        plt.figure()
        plt.xlabel("Number of KBest Features Used")
        plt.ylabel("Accuracy of classifier")
        scores = []
        for k in range(1, kvalues + 1):
            kbest = SelectKBest(k=k)
            scaled_K = kbest.fit_transform(features, labels)
            features_train, features_test, labels_train, labels_test = \
            train_test_split(scaled_K, labels, test_size = 0.3)
            scores.append(clf.fit(features_train, labels_train).score(features_test, labels_test))
        plt.plot(range(1, kvalues + 1), scores)
        plt.show()
